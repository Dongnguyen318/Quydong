# Bài 1: Cho trước 1 mảng số nguyên, yêu cầu tìm số lớn nhất, nhỏ nhất trong mảng và vị trí






        
    


        
var arr = [3, 6, 1, 30, 24, 54, 12];
var max = arr[0], min = arr[0];
var maxPosition, minPosition;

for (var i = 0; i < arr.length; i++) {
    if (max < arr[i]) {
        max = arr[i];
        maxPosition = i;
    }

    if (min > arr[i]) {
        min = arr[i];
        minPosition = i;
    }
}

console.log("Số lớn nhất là:" + max + " ở vị trí: " + maxPosition);
console.log("Số nhỏ nhất là:", min + " ở vị trí: " + minPosition);
