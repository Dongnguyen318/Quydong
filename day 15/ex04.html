# Bài 4: Kiểm tra số cùng dấu



    
        
    Input:
    
    Cho trước 2 số a, b
    
    Output:
    
    Kiểm tra 2 số cùng dấu hay không và hiển thị kết quả ra màn hình
    


        
var a = -8;
var b = -9;

console.log(`${a} ${b}`);

if (a == 0 || b == 0) console.log("Số 0 không phải số ấm cũng không phải số dương");
else if (a*b > 0) console.log("cùng dấu");
else console.log("Khác dấu");
